create table if not e user(
	id int auto_increment key,
    first_name char(255),
    last_name char(255),
    email char(255),
    password char(255),
    role text,
    address char(255),
    avatar text,
    phone char(255),
    discription text,
    sex int,
    facebook text,
    last_login date,
    expired boolean,
    locked boolean,
    enabled boolean,
    exp long,
    gold long,
    status int
);


create table if not exists game(
	id int auto_increment primary key,
    name text,
    content text,
    view int,
	release_time datetime,
    status int 
)engine=InnoDB default charset=utf8;

create table if not exists recordmatch(
	id int auto_increment primary key,
    start_time datetime,
    status int ,
	winner int,
    data text
)engine=InnoDB default charset=utf8;

create table if not exists recordusergame(
	id int auto_increment key,
	game_id int,
    match_id int,
	user_id int,
    opponent text,
    status int,
    foreign key(match_id) references recordmatch(id),
    foreign key(user_id) references user(id),
    foreign key(game_id) references game(id)
)engine=InnoDB default charset=utf8;


create table news(
	id int auto_increment key,
    title text,
    describes text,
    content text,
    user_id int,
    last_update datetime,
    time_create datetime,
    status int,
    view int,
    like_num int,
    comment text,
    background_image text,
    user_name text,
)

create table if not exists chat(
	id int auto_increment primary key,
    user_id int,
    message text,
	send_time datetime,
    status int ,
    foreign key(user_id) references user(id)
)engine=InnoDB default charset=utf8;
