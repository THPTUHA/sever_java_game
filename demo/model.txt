create table if not e user(
	id int auto_increment key,
    first_name char(255),
    last_name char(255),
    email char(255),
    password char(255),
    role text,
    address char(255),
    avatar text,
    phone char(255),
    description text,
    sex int,
    facebook text,
    last_login date,
    expired boolean,
    locked boolean,
    enabled boolean,
    exp long,
    gold long,
    status int
);


create table if not exists game(
	id int auto_increment primary key,
    name text,
    content text,
    view int,
	release_time datetime,
    status int 
)engine=InnoDB default charset=utf8;

create table if not exists recordusergame(
	id int auto_increment key,
	game_id int,
    start_time int,
	user_id_1 int,
    user_id_2 int,
    user_id_3 int,
    user_id_4 int,
    point_1 int,
    point_2 int,
    point_3 int,
    point_4 int,
    status int,
    winner int,
    data text,
    foreign key(user_id_1) references user(id),
    foreign key(user_id_2) references user(id),
    foreign key(user_id_3) references user(id),
    foreign key(user_id_4) references user(id),
    foreign key(game_id) references game(id)
)engine=InnoDB default charset=utf8;


create table news(
	id int auto_increment key,
    title text,
    describes text,
    content text,
    user_id int,
    last_update datetime,
    time_create datetime,
    status int,
    view int,
    like_num int,
    comment text,
    background_image text,
    user_name text,
)

create table if not exists chat(
	id int auto_increment primary key,
    user_id int,
    message text,
	send_time datetime,
    status int ,
    foreign key(user_id) references user(id)
)engine=InnoDB default charset=utf8;
