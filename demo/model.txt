create table user(
	id int auto_increment key,
    first_name char(255),
    last_name char(255),
    email char(255),
    password char(255),
    role text,
    address char(255),
    avatar text,
    phone char(255),
    discription text,
    sex int,
    facebook text,
    last_login date,
    expired boolean,
    locked boolean,
    enabled boolean,
    exp long,
    gold long,
    status int
);

create table news(
	id int auto_increment key,
    title text,
    describes text,
    content text,
    user_id int,
    last_update datetime,
    time_create datetime,
    status int,
    view int,
    like_num int,
    comment text,
    background_image text,
    user_name text,
)

create table play(
	id int auto_increment key,
	id_game int ,
	player text ,
    user_num int,
    winner int,
    status int,
    start_time datetime
);
